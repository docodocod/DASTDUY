    T=int(input())
    for test_case in range(1, T+1):
        N, M = map(int, input().split())
        arr = list(map(int, input().split()))
        sum_lst = []
        for i in range(N-M+1):
            sum = 0
            for j in range(i, i+M):
                sum += arr[j]
            sum_lst.append(sum)
    print(f'#{test_case}', max(sum_lst)-min(sum_lst))
